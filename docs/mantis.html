Mantis Robot SDK - Zenoh 二次开发接口

让客户无需安装 ROS2，通过 Zenoh 协议直接控制 Mantis 机器人。

Features:
  • 双臂控制：7自由度机械臂，支持单关节/多关节设置
  • 夹爪控制：左右夹爪开合
  • 头部控制：俯仰/偏航两自由度
  • 底盘控制：全向移动底盘
  • 仿真预览：RViz 实时预览（带平滑）
  • 关节限位：自动限制在安全范围内
Installation:

```bash



        pip install eclipse-zenoh
        pip install mantis-sdk


###### Quick Start:
> ```python



    from mantis import Mantis

    # 实机控制
    with Mantis(ip="192.168.1.100") as robot:
        robot.left_arm.set_shoulder_pitch(-0.5)
        robot.right_arm.set_joints([0.0] * 7)
        robot.head.look_left()
        robot.left_gripper.open()

    # 仿真预览
    with Mantis(sim=True) as robot:
        robot.left_arm.set_shoulder_pitch(-0.5)

223    def set_smoothing(self, alpha: float = 0.1, rate: float = 100.0):
224        """设置仿真模式的运动平滑参数。

226        使用 EMA（指数移动平均）算法进行平滑：

229        Args:

231                - 0.05: 非常平滑，响应慢

242                robot.connect()
243        """
244        self._sim_alpha = max(0.01, min(1.0, alpha))

249    def connect(self, timeout: float = 5.0, verify: bool = True) -> bool:
250        """连接到机器人。
251        
252        建立与机器人的 Zenoh 通信连接。实机模式下会验证机器人是否在线，
253        仿真模式下跳过验证直接连接。
254        
255        Args:
256            timeout: 连接超时时间（秒），默认 5.0
257            verify: 是否验证机器人在线，默认 True。
258                仿真模式下此参数被忽略。
259            
260        Returns:
261            bool: 连接是否成功
262        
263        Raises:
264            无异常抛出，失败时返回 False 并打印错误信息。
265        
266        Example:
267            .. code-block:: python
268            
269                robot = Mantis(ip="192.168.1.100")
270                if robot.connect():
271                    print("连接成功")
272                else:
273                    print("连接失败")
274        """
275        if self._connected:
276            return True
277        
278        # 仿真模式不验证
279        if self._sim_mode:
280            verify = False
281        
282        try:
283            config = zenoh.Config()
284            if self._router:
285                config.insert_json5("connect/endpoints", f'["{self._router}"]')
286            
287            self._session = zenoh.open(config)
288            
289            # 创建发布者
290            if self._sim_mode:
291                # 仿真模式：发布到 /joint_states（纯仿真，绕过 input_router）
292                self._publishers['sim_joints'] = self._session.declare_publisher(Topics.SIM_JOINT_STATES)
293            else:
294                # 实机模式：发布到控制话题
295                self._publishers['joints'] = self._session.declare_publisher(Topics.JOINT_CMD)
296                self._publishers['gripper'] = self._session.declare_publisher(Topics.GRIPPER)
297                self._publishers['head'] = self._session.declare_publisher(Topics.HEAD)
298                self._publishers['chassis'] = self._session.declare_publisher(Topics.CHASSIS)
299                self._publishers['pelvis'] = self._session.declare_publisher(Topics.PELVIS)
300            
301            # 验证机器人是否在线
302            if verify:
303                import time
304                received = []
305                
306                def _check_callback(sample):
307                    received.append(True)
308                
309                # 订阅反馈话题检测
310                sub = self._session.declare_subscriber(Topics.JOINT_FEEDBACK, _check_callback)
311                
312                # 等待消息
313                start = time.time()
314                while time.time() - start < timeout:
315                    if received:
316                        break
317                    time.sleep(0.1)
318                
319                sub.undeclare()
320                
321                if not received:
322                    self._session.close()
323                    self._session = None
324                    self._publishers.clear()
325                    target = self._router if self._router else "本机 (自动发现)"
326                    print(f"❌ 连接超时: 未检测到机器人 ({target})")
327                    print("   请检查:")
328                    print("   1) zenoh-bridge-ros2dds 是否已启动")
329                    print("   2) ROS2 节点是否在发布 /joint_states_fdb")
330                    print("   3) Domain ID 是否一致")
331                    return False
332            
333            self._connected = True
334            mode_str = "仿真预览" if self._sim_mode else "实机控制"
335            print(f"✅ 已连接到 Mantis 机器人 ({mode_str})")
336            
337            # 仿真模式：启动平滑发布线程
338            if self._sim_mode:
339                self._start_sim_smooth_thread()
340            
341            return True
342            
343        except Exception as e:
344            print(f"❌ 连接失败: {e}")
345            return False

347    def disconnect(self):
348        """断开与机器人的连接。
349        
350        停止所有运动，关闭 Zenoh 会话，释放资源。
351        如果正在运行平滑线程，也会一并停止。
352        
353        Note:
354            使用上下文管理器时会自动调用此方法。
355        """
356        if self._session:
357            # 停止平滑线程
358            if self._sim_thread is not None:
359                self._sim_running = False
360                self._sim_thread.join(timeout=1.0)
361                self._sim_thread = None
362            
363            # 停止运动
364            self._chassis.stop()
365            
366            for pub in self._publishers.values():
367                pub.undeclare()
368            for sub in self._subscribers.values():
369                sub.undeclare()
370            
371            self._session.close()
372            self._session = None
373            self._publishers.clear()
374            self._subscribers.clear()
375            self._connected = False
376            print("✅ 已断开连接")

380    def _start_sim_smooth_thread(self):
381        """启动仿真平滑发布线程。
382        
383        在仿真模式连接成功后自动调用，启动后台线程进行关节位置平滑插值。
384        """
385        if self._sim_thread is not None:
386            return
387        
388        self._sim_running = True
389        self._sim_thread = threading.Thread(target=self._sim_smooth_loop, daemon=True)
390        self._sim_thread.start()

392    def _sim_smooth_loop(self):
393        """平滑插值循环。
394        
395        使用 EMA 算法逐渐将当前关节状态逼近目标状态，
396        并以固定频率发布到仿真话题。
397        """
398        interval = 1.0 / self._sim_frequency
399        
400        while self._sim_running:
401            # EMA 平滑：current = current + alpha * (target - current)
402            for name in ALL_URDF_JOINTS:
403                current = self._sim_current_states[name]
404                target = self._sim_target_states[name]
405                self._sim_current_states[name] = current + self._sim_alpha * (target - current)
406            
407            # 发布平滑后的状态
408            self._publish_sim_state()
409            
410            time.sleep(interval)

423    def _publish_joints(self):
424        """发布手臂关节角度。
425        
426        将左右臂的关节位置发送到机器人或仿真环境。
427        """
428        self._check_connection()
429        
430        # 合并左右臂位置
431        positions = self._left_arm._positions + self._right_arm._positions
432        
433        if self._sim_mode:
434            # 仿真模式：更新目标关节状态（由平滑线程逐渐逼近并发布）
435            for i, (serial_name, pos) in enumerate(zip(JOINT_NAMES, positions)):
436                urdf_name = SERIAL_TO_URDF_MAP.get(serial_name)
437                if urdf_name:
438                    direction = JOINT_DIRECTION_MAP.get(serial_name, 1)
439                    self._sim_target_states[urdf_name] = pos * direction
440            # 不再手动调用 _publish_sim_state，由平滑线程处理
441        else:
442            # 实机模式：直接发送
443            payload = CDREncoder.encode_joint_state(
444                names=JOINT_NAMES,
445                positions=positions
446            )
447            self._publishers['joints'].put(payload)
